```dockerfile
WORKDIR /app

COPY package*.json ./

RUN npm ci

# Bundle app source inside Docker image
COPY . .

# Build the frontend application
RUN npm run build
```

```yaml

  backend:
    image: mern-blog-app-backend:latest
    build: ./backend
    ports:
      - "3000:3000"
    env_file:
      - ./backend/.env

  frontend:
    image: mern-blog-app-frontend:latest
    build: ./frontend
    ports:
      - "5173:80"

  mongo:
    image: mongo:latest
    ports:
      - "27017:27017"
    volumes:
      - mongo-data:/data/db

  mongo-data:
```
